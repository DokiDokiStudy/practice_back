# 애플리케이션 빌드 실전 예제:Nodejs 소스 코드

```
FROM diamol/node AS builder

WORKDIR /src
COPY src/package.json .  // package.json을 가져온다.
RUN npm install // 의존성 설치 

# app
FROM diamol/node

EXPOSE 80
CMD ["node", "server.js"]

WORKDIR /app
COPY --from=builder /src/node_modules/ /app/node_modules/ // 설치된 의존성 파일을 작업 디렉토리로 가져온다
COPY src/ . // 호스트에서 작업 디렉토리로 소스 코드를 가져온다.
```
